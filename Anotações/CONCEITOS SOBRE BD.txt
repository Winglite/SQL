-- SELECAO, PROJECAO E JUNCAO

--PROJECAO ---> E TUDO QUE VOCE QUER VER NA TELA.

SELECT NOME, SEXO, EMAIL --> PROJEÇÃO


--SELECAO ---> E UM SUBCONJUNTO DO CONJUNTO TOTAL DE REGISTROS DE UMA TABELA 
(CLAUSULA DE SELECAO -> WHERE)

SELECT NOME, SEXO, EMAIL --> PROJEÇÃO
FROM CLIENTE --> ORIGEM
WHERE SEXO = 'F' --> SELECAO


--JUNCAO (JOIN) --> E A JUNCAO DOS DADOS DE DUAS OU MAIS TABELAS

INNER JOIN (JUNCAO DE DENTRO - TUDO QUE E COMUM ENTRE AS TABELAS)

SELECT NOME, SEXO, BAIRRO, CIDADE --> PROJECAO 
FROM CLIENTE --> ORIGEM
INNER JOIN ENDERECO --> JUNCAO
ON IDCLIENTE = ID_CLIENTE
WHERE SEXO = 'F'; --> SELECAO

    --
    --DDL - DATA DEFINITION LANGUAGE (COMANDOS UTILIZADOS)

CRIAR TABELA --> 
                   CREATE TABLE PRODUTO(
	             IDPRODUTO INT PRIMARY KEY AUTO_INCREMENT,
	             NOME_PRODUTO VARCHAR(30) NOT NULL,
	             PRECO INT,
	             FRETE FLOAT(10,2) NOT NULL
                 );


ALTERAR O NOME DO ATRIBUTO E TIPO -->
               ALTER TABLE PRODUTO
               CHANGE PRECO VALOR_UNITARIO INT NOT NULL;

OU
              -- USAR O CHANGE PARA ALTERAR O TIPO, REPETE O NOME DO ATRIBUTO

              ALTER TABLE PRODUTO
              CHANGE VALOR_UNITARIO VALOR UNITARIO INT;

OU

              -- USAR O MODIFY PARA ALTERAR O TIPO, NAO REPETE O NOME DO ATRIBUTO

             ALTER TABLE PRODUTO
             MODIFY VALOR_UNITARIO VARCHAR(50) NOT NULL;


ADICIONAR UMA COLUNA (CASO ERRE O IDEAL E DAR DROP NA TABELA E FAZER NOVAMENTE) -->

              ALTER TABLE PRODUTO
              ADD PESO FLOAT(10,2) NOT NULL;


ADICIONAR COLUNA EM ORDEM ESPECIFICA -->

             ALTER TABLE PRODUTO
             ADD COLUMN PESO FLOAT(10,2) NOT NULL
             AFTER NOME_PRODUTO;

OU
             ALTER TABLE PRODUTO
             ADD COLUMN PESO FLOAT(10,2) NOT NULL
             FIRST;


APAGAR UMA COLUNA -->

             ALTER TABLE PRODUTO
             DROP COLUMN PESO;
